        <script>
            document.getElementById('add-song-form').addEventListener('submit', async (e) => {
                e.preventDefault();
                const form = e.target;
                const data = {
                    titulo: form.titulo.value,
                    artista: form.artista.value,
                    album: form.album.value,
                    caminho_arquivo: form.caminho.value
                };
                
                // Simulação de envio (substitua pela URL real do seu backend)
                try {
                    const response = await fetch('http://localhost/player-de-musica/src/create_song.php', {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify(data)
                    });
                    
                    if (response.ok) {
                        const result = await response.json();
                        alert(result.mensagem || 'Música adicionada com sucesso!');
                        form.reset(); // Limpa o formulário após o envio
                        
                        // Aqui você pode adicionar a nova música à lista
                        // Por simplicidade, apenas recarregamos a página
                        // Em um projeto real, você adicionaria dinamicamente
                        location.reload();
                    } else {
                        alert('Erro ao adicionar música. Tente novamente.');
                    }
                } catch (error) {
                    console.error('Erro:', error);
                    alert('Erro de conexão. Verifique se o servidor está rodando.');
                }
            });
        </script>
